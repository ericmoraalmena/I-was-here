import pygame
import sys

# Inicializar Pygame
pygame.init()

# Definir el tamaño de la pantalla
ANCHO, ALTO = 800, 600
pantalla = pygame.display.set_mode((ANCHO, ALTO))
pygame.display.set_caption("Juego con pantalla negra")

# Definir colores
NEGRO = (0, 0, 0)
BLANCO = (255, 255, 255)

# Función principal del juego
def main():
    
    pixel_menu_x = 430
    pixel_menu_y = 450
    
    text_pos_p_continuar= (pixel_menu_x, pixel_menu_y)
    
    pixel_menu2_x = 430
    pixel_menu2_y = 480
    
    text_pos_p_continuar2= (pixel_menu2_x, pixel_menu2_y)
    
    while True:
        pantalla.fill(NEGRO)
        fuente_p = pygame.font.Font("fuentes/gamefont.otf", 70)
        
        text_p_continuar = fuente_p.render("pulsa A o >-- para", True, BLANCO)
        text_rect_p_continuar = text_p_continuar.get_rect(center=text_pos_p_continuar)
        
        pantalla.blit(text_p_continuar, text_rect_p_continuar)
        
        text_p_continuar2 = fuente_p.render("la opción 1", True, BLANCO)
        text_rect_p_continuar2 = text_p_continuar2.get_rect(center=text_pos_p_continuar2)
        
        pantalla.blit(text_p_continuar2, text_rect_p_continuar2)
        # Manejo de eventos
        for event in pygame.event.get():
            if event.type == pygame.QUIT:
                pygame.quit()
                sys.exit()

        # Actualizar pantalla
        
        pygame.display.flip()

# Llamar a la función principal
if __name__ == "__main__":
    main()
